    "## 1. El Laplaciano Magn\u00e9tico $L^{(q)}$\n",
    "La fuga de informaci\u00f3n (data leakage) en atenci\u00f3n sim\u00e9trica se rompe introduciendo una fase magn\u00e9tica $q$.\n"
   ]
  },
  {
   "cell_type": "code",
   "execution_count": null,
   "metadata": {},
   "outputs": [],
   "source": [
    "N = 10\n",
    "q = math.pi / 4.0\n",
    "\n",
    "H_q = np.zeros((N, N))\n",
    "deg = np.zeros(N)\n",
    "\n",
    "for i in range(1, N):\n",
    "    j = i - 1\n",
    "    H_q[j, i] = math.cos(q)\n",
    "    H_q[i, j] = math.cos(2*q)\n",
    "    deg[i] += 1\n",
    "    deg[j] += 1\n",
    "\n",
    "L0 = np.diag(deg) - H_q\n",
    "\n",
    "plt.imshow(L0, cmap=\"coolwarm\")\n",
    "plt.title(\"Laplaciano Magn\u00e9tico (q=pi/4)\")\n",
    "plt.colorbar()\n",
    "plt.show()\n",
    "\n",
    "print(\"Notar la asimetr\u00eda: El flujo hacia adelante tiene m\u00e1s peso que el flujo hacia atr\u00e1s, introduciendo causalidad estricta.\")\n"
   ]
